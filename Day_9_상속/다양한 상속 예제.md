# 자바 상속 구조 예제 3개

## 예제 1: 기본적인 동물 상속 구조

```java
// 부모 클래스
class Animal {
    private String name;
    
    public Animal(String name) {
        this.name = name;
    }
    
    public void eat() {
        System.out.println(name + "이(가) 먹이를 먹습니다.");
    }
    
    public void sleep() {
        System.out.println(name + "이(가) 잠을 잡니다.");
    }
    
    public String getName() {
        return name;
    }
}

// 자식 클래스 1
class Dog extends Animal {
    public Dog(String name) {
        super(name); // 부모 클래스의 생성자 호출
    }
    
    // 메소드 오버라이딩
    @Override
    public void eat() {
        System.out.println(getName() + "이(가) 사료를 맛있게 먹습니다.");
    }
    
    // 새로운 메소드 추가
    public void bark() {
        System.out.println(getName() + ": 왈왈!");
    }
}

// 자식 클래스 2
class Cat extends Animal {
    public Cat(String name) {
        super(name);
    }
    
    @Override
    public void eat() {
        System.out.println(getName() + "이(가) 생선을 조심스럽게 먹습니다.");
    }
    
    public void meow() {
        System.out.println(getName() + ": 야옹~");
    }
}

// 메인 클래스
public class AnimalTest {
    public static void main(String[] args) {
        Dog dog = new Dog("뽀삐");
        Cat cat = new Cat("나비");
        
        dog.eat();    // 오버라이딩된 메소드 호출
        dog.sleep();  // 부모 클래스 메소드 호출
        dog.bark();   // 자식 클래스 고유 메소드 호출
        
        System.out.println(); // 줄바꿈
        
        cat.eat();
        cat.sleep();
        cat.meow();
    }
}
```

## 예제 2: 도형 계산기 상속 구조

```java
// 부모 클래스
class Shape {
    private String color;
    
    public Shape(String color) {
        this.color = color;
    }
    
    public double getArea() {
        System.out.println("도형의 넓이를 계산합니다.");
        return 0;
    }
    
    public void draw() {
        System.out.println(color + "색 도형을 그립니다.");
    }
    
    public String getColor() {
        return color;
    }
}

// 자식 클래스 1
class Circle extends Shape {
    private double radius;
    
    public Circle(String color, double radius) {
        super(color);
        this.radius = radius;
    }
    
    @Override
    public double getArea() {
        return Math.PI * radius * radius;
    }
    
    @Override
    public void draw() {
        System.out.println(getColor() + "색 원을 그립니다. 반지름: " + radius);
    }
}

// 자식 클래스 2
class Rectangle extends Shape {
    private double width;
    private double height;
    
    public Rectangle(String color, double width, double height) {
        super(color);
        this.width = width;
        this.height = height;
    }
    
    @Override
    public double getArea() {
        return width * height;
    }
    
    @Override
    public void draw() {
        System.out.println(getColor() + "색 직사각형을 그립니다. 너비: " + width + ", 높이: " + height);
    }
}

// 메인 클래스
public class ShapeTest {
    public static void main(String[] args) {
        Shape circle = new Circle("빨강", 5.0);
        Shape rectangle = new Rectangle("파랑", 4.0, 6.0);
        
        // 다형성 예시
        Shape[] shapes = {circle, rectangle};
        
        for (Shape shape : shapes) {
            shape.draw();
            System.out.println("넓이: " + shape.getArea());
            System.out.println();
        }
    }
}
```

## 예제 3: 직원 관리 시스템

```java
// 부모 클래스
class Employee {
    private String name;
    private int employeeId;
    private double baseSalary;
    
    public Employee(String name, int employeeId, double baseSalary) {
        this.name = name;
        this.employeeId = employeeId;
        this.baseSalary = baseSalary;
    }
    
    public double calculateSalary() {
        return baseSalary;
    }
    
    public void work() {
        System.out.println(name + "이(가) 업무를 수행합니다.");
    }
    
    // Getter 메소드들
    public String getName() {
        return name;
    }
    
    public int getEmployeeId() {
        return employeeId;
    }
    
    public double getBaseSalary() {
        return baseSalary;
    }
}

// 자식 클래스 1: 관리자
class Manager extends Employee {
    private double bonus;
    
    public Manager(String name, int employeeId, double baseSalary, double bonus) {
        super(name, employeeId, baseSalary);
        this.bonus = bonus;
    }
    
    @Override
    public double calculateSalary() {
        return getBaseSalary() + bonus;
    }
    
    @Override
    public void work() {
        System.out.println(getName() + " 매니저가 팀을 관리합니다.");
    }
    
    public void conductMeeting() {
        System.out.println(getName() + " 매니저가 회의를 주관합니다.");
    }
}

// 자식 클래스 2: 개발자
class Developer extends Employee {
    private String programmingLanguage;
    private int overtimeHours;
    
    public Developer(String name, int employeeId, double baseSalary, 
                     String programmingLanguage, int overtimeHours) {
        super(name, employeeId, baseSalary);
        this.programmingLanguage = programmingLanguage;
        this.overtimeHours = overtimeHours;
    }
    
    @Override
    public double calculateSalary() {
        // 시간외 수당: 시간당 기본급의 1.5배
        double overtimePay = (getBaseSalary() / 160) * 1.5 * overtimeHours;
        return getBaseSalary() + overtimePay;
    }
    
    @Override
    public void work() {
        System.out.println(getName() + " 개발자가 " + programmingLanguage + "로 코드를 작성합니다.");
    }
    
    public void debugCode() {
        System.out.println(getName() + " 개발자가 코드 디버깅을 합니다.");
    }
}

// 메인 클래스
public class EmployeeTest {
    public static void main(String[] args) {
        Employee manager = new Manager("김철수", 1001, 5000000, 1000000);
        Employee developer = new Developer("이영희", 1002, 4000000, "Java", 20);
        
        // 각 직원 정보 출력
        Employee[] employees = {manager, developer};
        
        for (Employee emp : employees) {
            emp.work();
            System.out.println("이름: " + emp.getName());
            System.out.println("사원번호: " + emp.getEmployeeId());
            System.out.println("월급: " + emp.calculateSalary() + "원");
            
            // 타입 체크 후 특정 메소드 호출
            if (emp instanceof Manager) {
                ((Manager) emp).conductMeeting();
            } else if (emp instanceof Developer) {
                ((Developer) emp).debugCode();
            }
            
            System.out.println("----------------------");
        }
    }
}
```

## 주요 개념 설명

1. **상속 (Inheritance)**: 자식 클래스가 부모 클래스의 속성과 메소드를 물려받음
2. **메소드 오버라이딩 (Method Overriding)**: 자식 클래스에서 부모 클래스의 메소드를 재정의
3. **super 키워드**: 부모 클래스의 생성자나 메소드 호출
4. **다형성 (Polymorphism)**: 부모 타입으로 자식 객체 참조 가능
5. **instanceof 연산자**: 객체의 실제 타입 확인
